    /**
 * Module to handle MongoDB connection
 * @module helpers/MongoConnection
 */

let URL;
let OPTION;
let CLIENT;
let ObjectID = require('mongodb').ObjectID;

const MongoClient = require('mongodb').MongoClient;
const assert = require('assert');
const mongoose = require('mongoose');


let DB = null;

/**
 * Initialize mongoDB with connection to MongoDB and database selection.
 * @param  {string} dbname - Database name that will be used.
 * @param  {Object} option - Unused for now...
 */
function init(dbname, option = {}) {
	return new Promise((res, rej) => {
		CLIENT.connect(function (err) {
			if (err)
				return rej(err);

			DB = CLIENT.db(dbname);
			res(DB);
		});
	})
}

function parseID(id) {
	if (mongoose.Types.ObjectId.isValid(id) && typeof id === "object") {
		return id;
	}
	else {
		try {
			id = ObjectID(id);
		}
		catch (e) {
			throw new Error("Id is not a valid id");
		}
		return id;
	}
}

function getCollection(db) {
	const collection = DB.collection(db);
	return (collection);
}

module.exports = (url, option) => {
	const client = new MongoClient(url, { ...option, ...{ useUnifiedTopology: true } });
	CLIENT = client;
	return ({
		init: init,
		db: () => DB,
		parseID: parseID,
		col: getCollection,
	});
}
